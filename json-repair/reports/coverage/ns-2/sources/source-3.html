 import java.util.Arrays;

 public enum KeySymbol {

     R_BRACE("}"),

     R_BRACKET("]"),
     COMMA(","),
     COLON(":"),
     STRING("STRING"),
     NUMBER("NUMBER"),
     TRUE("true"),
     FALSE("false"),
     NULL("null"),
     EOF("<EOF>"),
     TOKEN("token");
     
     private String val;
     
     KeySymbol(String val) {
         this.val = val;
     }
     
     public String val() {
         return val;
     }

                 L_BRACE.val, R_BRACE.val, L_BRACKET.val, R_BRACKET.val,
                 COLON.val, COMMA.val);
     }
 }
